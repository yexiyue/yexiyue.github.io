## v-model

默认情况下，组件上的 `v-model` 使用 `modelValue` 作为 prop 和 `update:modelValue` 作为事件。我们可以通过向 `v-model` 传递参数来修改这些名称：

```vue
//子组件
<template>
  <div v-if="modelValue">
    <div>主题
    <button @click="change">关闭</button>
    </div>
    <div>
      内容{{modelValue}}
    </div>
  </div>
</template>

<script setup lang="ts">
defineProps<{
  modelValue:any
}>()

const emit=defineEmits<{
  (e:'update:modelValue',flag:boolean):void
}>()
const change=()=>{
  emit('update:modelValue',false)
}

</script>

<style scoped lang="less">
div {
  border:1px solid black;
  background-color: rgb(160, 46, 31);
  div:first-child {
    height: 30px;
    width: 300px;
    text-align: center;
    border-bottom: 1px solid black;
  }
  div:last-child {
    height: 300px;
    width: 300px;
  }
}
</style>
```

### 2.定义插件

```typescript
import {App ,createVNode,VNode,render ,ComponentCustomProperties} from 'vue'
import Loading from './loading.vue'

//声明文件
declare module '@vue/runtime-core' {
  export interface ComponentCustomProperties {
    $loading:{
      show():void,
      unShow():void
    }
  }
}

export default {
  //这是插件默认加载函数
  install(app:App){
    //1.把组件转换为虚拟dom
    const vnode:VNode=createVNode(Loading)
    //2.转真实dom需要用render函数
    render(vnode,document.body)
    //3.把组件的方法挂载到全局
     /**
      * console.log(vnode.component?.exposed)
      * 在组件通过defineExpose把方法暴露出来
      */
     //挂载到全局
     app.config.globalProperties.$loading={
       show:vnode.component?.exposed?.show,
       unShow:vnode.component?.exposed?.unShow
     }
  }
}
```

### 3.编写pinia插件做本地持久化

```typescript
import {PiniaPluginContext} from 'pinia'
import {toRaw} from 'vue'
/**
 * @function setStorage
 * @param key 传入键名
 * @param value 任意值
 * @returns 没有返回值
 */
const setStorage=(key:string,value:any)=>{
  localStorage.setItem(key,JSON.stringify(value))
}
/**
 * 
 * @param key 键名
 * @returns 返回缓存的对象
 */
const getStorage=(key:string):object=>{
  return localStorage.getItem(key) ? JSON.parse(localStorage.getItem(key) as string) : {}
}
export const piniaPlugin=(options:{key:string})=>{

  return (context:PiniaPluginContext)=>{
    //结构store
    const {store} =context
    
    //首次刷新读取缓存
    const data=getStorage(`${options?.key ?? 'yeshifu'}-${store.$id}`)
    /* console.log(data); */
    
    //每次修改都触发保存
    store.$subscribe(()=>{
      /* console.log(`${options?.key ?? 'yeshifu'}-${store.$id}`,toRaw(store.$state)) */
      setStorage(`${options?.key ?? 'yeshifu'}-${store.$id}`,toRaw(store.$state))
    })

    //直接返回值相当于修改store的状态
    return {
      ...data
    }
  }

}
```

```ts
//导入pinia模块
import {createPinia} from 'pinia'
const store=createPinia()
//导入pinia插件 
import {piniaPlugin} from './store/pinia_plugin'
store.use(piniaPlugin({
  key:'pinia'
}))
```
